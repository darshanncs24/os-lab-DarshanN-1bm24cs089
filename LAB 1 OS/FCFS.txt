#include<stdio.h>
void main()
{
    int n, i, j;
    printf("Enter the number of processes:");
    scanf("%d",&n);
    int P[n], AT[n], BT[n], CT[n], TAT[n], WT[n];
        int curr=0;
     double  atat=0;
      double awt=0;
    for(i=0;i<n;i++)
    {
        printf("Enter the process id, arrival time and burst time: ");
        scanf("%d%d%d",&P[i],&AT[i],&BT[i]);
    }

     for(i=0;i<n;i++)
    {
       for(j=0;j<n-1;j++)
        {
        if(AT[j]>AT[j+1])
            {

                int temp=AT[j+1];
                AT[j+1]=AT[j];
                AT[j]=temp;

                temp=BT[j+1];
                BT[j+1]=BT[j];
                BT[j]=temp;

                 temp=P[j+1];
                P[j+1]=P[j];
                P[j]=temp;

            }

        }

    }

     for(i=0;i<n;i++)
    {if (curr<AT[i])
    curr=AT[i];
    CT[i]=curr+BT[i];
    TAT[i]=CT[i]-AT[i];
    WT[i]=TAT[i]-BT[i];
    curr=CT[i];
    atat+=TAT[i];
    awt+=WT[i];
    }
    printf("Process_id AT BT CT TAT WT\n");
   for (i=0;i<n;i++)
   {
       printf("%d\t%d\t%d\t%d\t%d\t%d\n",P[i],AT[i],BT[i],CT[i],TAT[i],WT[i]);

       }
       printf("average tat %lf\n", (atat/n));
         printf("average wt %lf\n", (awt/n));

       }
